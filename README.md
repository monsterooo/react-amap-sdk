## Install

```bash
npm install --save react-amap-sdk
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-amap-sdk'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
